       ```
       private Timer poiDensityDebounceTimer;
       
       private void setupPOIDensityListeners() {
         view.getDungeonDensitySlider().valueProperty().addListener((obs, oldVal, newVal) -> {
           if (poiDensityDebounceTimer != null) {
             poiDensityDebounceTimer.cancel();
           }
           poiDensityDebounceTimer = new Timer();
           poiDensityDebounceTimer.schedule(new TimerTask() {
             @Override

             }
           }, 300);
         });

       }
       ```

       ```
       private void triggerMapRegeneration() {

         double settlementDensity = view.getSettlementDensitySlider().getValue();
         
         model.generateMap(
           currentSeed, currentSize, ..., 
           dungeonDensity, landmarkDensity, settlementDensity
         );
         render();
       }
       ```

    3. Ensure MainModel.generateMap() passes density parameters to MapGenerator.generate().
